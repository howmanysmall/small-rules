## Rule Details

## Examples

<RuleExample type="fail" title="Fails">

```ts
class MyClass {
	public value = 0;
	public incrementBy(amount: number): void {
		const value = this.value + amount;
		this.value = value;
		this.notifyChanges(value);
	}

	private notifyChanges(value: number): void {
		console.log(`Value changed to ${value}`);
	}
}
```

</RuleExample>

<RuleExample type="pass" title="Passes">

```ts
function notifyChanges(value: number): void {
	console.log(`Value changed to ${value}`);
}

class MyClass {
	public value = 0;
	public incrementBy(amount: number): void {
		const value = this.value + amount;
		this.value = value;
		notifyChanges(value);
	}
}
```

</RuleExample>
